    26тЖТ    26тЖТ        return Object.assign({}, p, { _id: 'seed_' + i, published: true, sortOrder: i });
    27тЖТ    27тЖТ      });

   698тЖТ   698тЖТ    var e = document.createElement(tag);
   699тЖТ   699тЖТ    if (attrs) {
   700тЖТ   700тЖТ      Object.keys(attrs).forEach(function (k) {
   701тЖТ   701тЖТ        if (k === 'className') e.className = attrs[k];
   702тЖТ   702тЖТ        else if (k === 'style' && typeof attrs[k] === 'object') Object.assign(e.style, attrs[k]);
   703тЖТ   703тЖТ        else e.setAttribute(k, attrs[k]);
   704тЖТ   704тЖТ      });
   705тЖТ   705тЖТ    }
   706тЖТ   706тЖТ    if (Array.isArray(textOrChildren)) {
   707тЖТ   707тЖТ      textOrChildren.forEach(function (c) { e.appendChild(c); });
   708тЖТ   708тЖТ    } else if (textOrChildren !== undefined) {
   709тЖТ   709тЖТ      e.textContent = textOrChildren;
   710тЖТ   710тЖТ    }
   711тЖТ   711тЖТ    return e;
